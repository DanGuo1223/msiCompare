Model-based simulation

conditions <- ifelse((expand.grid(x=1:30, y=1:30)$x %in% (1+floor(30/5)):(30-floor(30/5)) & expand.grid(x=1:30, y=1:30)$y %in% (1+floor(30/5)):(30-floor(30/5))), 2, 1)

 s <- simSingle(
      reps = 3,
      diff = log2(1.5),
      tau2 = 0.1,
      sig2 = 0.1,
      seed = 8372,
      size1 = 30,
      center.pattern = T,
      pattern = conditions
      )
summary(s)
